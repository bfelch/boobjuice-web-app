CREATE TABLE IF NOT EXISTS `PUMP_PROFILE` (
  `U_PROFILE` smallint(6) NOT NULL AUTO_INCREMENT,
  `T_PROFILE` varchar(50) NOT NULL,
  `T_DESCRIPTION` varchar(500) DEFAULT NULL,
  `C_COLOR_HEX` char(6) DEFAULT NULL,
  PRIMARY KEY (`U_PROFILE`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_general_ci;
